cd test  - переходим в папку test (есди это папка и она существует)
cd ..    - переход в родительскую папку 
cd .     - переход в эту же папку (мы никуда не перейдем)
cd       - переход в домашнюю папку (~)
cd~      - переход в домашнюю папку
pwd      - показывает абсалютный путь до текущей папки
mkdir test - создание новой папки test
mkdir test/new_dir  -создание новой папки new_dir в уже существующей папке test
touch test.txt  - создание файл test.txt
rm test.txt  - удаление файла test.txt
rm -rf test    - удаление папки test со всем её содержимым
sudo rm -rf/ - никогда не использовать!
ls test   - просмотр всех файлов и папок внутри папки test
ls -a test   - просмотр всех файлов и папок внутри папки test включая скрытые
к любой команде можно дописать  --help и выйдет описание как исползовать эту команду 
ls -a  - просмотр всех файлов и папок с более подробной информацией

nano test.txt  - открыть файл в редакторе nano
чтобы сохранить (ctrl s)
чтобы выйти (ctrl x)
